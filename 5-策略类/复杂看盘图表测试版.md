def ticks_(records, k):
    if len(records) == 0:
        return []
    if isinstance(records[0], int) or isinstance(records[0], float):
        return records

    ticks = [None] * len(records)
    for i in range(len(records)):
        ticks[i] = records[i][k]
        return ticks
